# Fix column name and recreate CSV
python3 -c "
import pandas as pd
df = pd.read_csv('certs_fixed.csv', dtype=str)

# Strip whitespace from ALL column names
df.columns = df.columns.str.strip()

print('Cleaned columns:', df.columns.tolist())
print('TWX_AppKey values:', df['TWX_AppKey'].tolist())

df.to_csv('certs_fixed.csv', index=False)
print('Saved')
"
