# Convert tab-separated to comma CSV and fix all issues
python3 -c "
import pandas as pd

# Read tab-separated original, skip the bad index column
df = pd.read_csv('certs.csv', sep='\t', dtype=str)

# Drop unnamed index column if present
df = df.loc[:, ~df.columns.str.contains('^Unnamed')]

print('Columns:', df.columns.tolist())
print('Rows:', len(df))
print(df.to_string())

# Save without index
df.to_csv('certs_fixed.csv', index=False)
print('Saved certs_fixed.csv')
"


kubectl create configmap cert-data-file --from-file=certs.csv=certs_fixed.csv

  --type merge \
