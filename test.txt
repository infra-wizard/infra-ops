- name: Archive Executable Artifacts
  uses: actions/upload-artifact@v4
  if: always()
  with:
    name: Executable
    path: |
      **/ALQAgentExecutable*.zip
    retention-days: 30
    if-no-files-found: warn

- name: Archive Source Artifacts
  uses: actions/upload-artifact@v4
  if: always()
  with:
    name: Source
    path: |
      **/ALQAgentSource*.zip
    retention-days: 30
    if-no-files-found: warn
