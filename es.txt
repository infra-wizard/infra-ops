# ðŸ‘‡ argument parsing block goes right after the function
while [[ $# -gt 0 ]]; do
  case "$1" in
    --deploy)
      deploy
      shift
      ;;
    --regcode)
      export REGCODE="$2"
      shift 2
      ;;
    *)
      echo "Unknown option: $1"
      exit 1
      ;;
  esac
done

# ðŸ‘‡ now you can use REGCODE safely
if [ -z "$REGCODE" ]; then
  echo "Error: --regcode is required"
  exit 1
fi
