

# For RHEL/CentOS
sudo rpm -Uvh https://packages.microsoft.com/config/rhel/8/packages-microsoft-prod.rpm
sudo yum install blobfuse

# For Arch (use AUR)
yay -S blobfuse

# Create config file
sudo mkdir -p /etc/blobfuse
sudo nano /etc/blobfuse/connection.cfg
```

Add this content:
```
accountName <your-storage-account-name>
accountKey <your-storage-account-key>
containerName <your-container-name>
```

sudo chmod 600 /etc/blobfuse/connection.cfg

sudo mkdir -p /mnt/blobstorage
sudo mkdir -p /mnt/blobfusetmp

sudo blobfuse /mnt/blobstorage --tmp-path=/mnt/blobfusetmp --config-file=/etc/blobfuse/connection.cfg -o attr_timeout=240 -o entry_timeout=240 -o negative_timeout=120 -o allow_other
