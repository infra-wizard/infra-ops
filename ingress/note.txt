- name: Create new branch
      run: |
        BRANCH_NAME=$(date +"V%Y-%m-%d")
        git checkout -b $BRANCH_NAME
        git config --global user.email "${{ github.event.pusher.email }}"
        git config --global user.name "${{ github.triggering_actor }}"
        sed -i 's/tag: .*/tag: "${{ github.event.inputs.docker_tag }}"/' app/*/values.yaml
        git add *
        git commit -m "updating values"
        git push origin $BRANCH_NAME
        gh pr create --base main --head $BRANCH_NAME --title "$BRANCH_NAME" --body "QF Prod Deploy"

cd /tmp
tar -xzf gh.tar.gz

# Install to /usr/local/bin (system-wide, no package manager needed)
sudo mv gh_2.67.0_linux_amd64/bin/gh /usr/local/bin/gh
sudo chmod +x /usr/local/bin/gh

# Optional: install man pages
sudo mkdir -p /usr/local/share/man/man1
sudo cp gh_2.67.0_linux_amd64/share/man/man1/* /usr/local/share/man/man1/
